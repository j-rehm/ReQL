CREATE TABLE 'appointments' (patient_name, doctor_name, apt_date, apt_time, topic) : line format /([A-Za-z\.\- ]+);([A-Za-z\.\- ]+);((?:\d{1,2}\/){2}\d{4}) (\d{1,2}:\d{2});(.*)/ file 'src/main/resources/appts.txt';
